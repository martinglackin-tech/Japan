        // --- DOM ELEMENTS ---

        const mainContent = document.getElementById('main-content');

        const navLearn = document.getElementById('nav-learn');

        // --- AUDIO FUNCTIONALITY ---

        function loadVoices() {

            if ('speechSynthesis' in window) {

                japaneseVoices = window.speechSynthesis.getVoices().filter(voice => voice.lang === 'ja-JP');

            }

        }


        function playAudio(text, event) {

            if (event) event.stopPropagation();

            if ('speechSynthesis' in window) {

                // Cancel any previous speech

                window.speechSynthesis.cancel();


                const utterance = new SpeechSynthesisUtterance(text);

                utterance.lang = 'ja-JP';

                utterance.rate = 0.9;

                

                // If Japanese voices are loaded, use the first available one

                if (japaneseVoices.length > 0) {

                    utterance.voice = japaneseVoices[0];

                } else {

                    // This is a fallback if voices haven't loaded for some reason

                    console.warn("Japanese voice not found, using browser default for ja-JP.");

                }

                

                window.speechSynthesis.speak(utterance);

            } else {

                console.error('Text-to-Speech not supported in this browser.');

            }

        }

            </svg>`;

        }

                            <div class="flex items-center mt-2">

                                <p class="jp-font text-gray-900 text-2xl">${phrase.japanese}</p>

                                <button class="play-audio-btn text-gray-500 hover:text-pink-500 ml-4" data-text="${phrase.japanese}">${speakerIconSVG}</button>

                            </div>

            mainContent.innerHTML = content;


            mainContent.querySelectorAll('.play-audio-btn').forEach(button => {

                button.addEventListener('click', (e) => playAudio(e.currentTarget.dataset.text, e));

            });

        }


        function showFlashcardsView() {

            currentFlashcardIndex = 0;

            renderFlashcard();

        }


        function renderFlashcard() {

            const phrase = phrases[currentFlashcardIndex];

        function showQuizView() {

            setActiveNav(navQuiz);

            startQuiz();

        }

        

        function startQuiz() {

            quizScore = 0;

            currentQuizQuestionIndex = 0;

            quizQuestions = generateQuizQuestions();

            renderQuizQuestion();

        }


        function generateQuizQuestions() {

            const shuffledPhrases = [...phrases].sort(() => 0.5 - Math.random());

            return shuffledPhrases.map(correctPhrase => {

                const wrongPhrases = [...phrases]

                    .filter(p => p.id !== correctPhrase.id)

                    .sort(() => 0.5 - Math.random())

                    .slice(0, 3);

                

                const options = [correctPhrase, ...wrongPhrases].sort(() => 0.5 - Math.random());

                return {

                    question: correctPhrase.english,

                    options: options.map(p => ({ romaji: p.romaji, japanese: p.japanese })),

                    answer: correctPhrase.romaji

                };

            });

        }


        function renderQuizQuestion() {

            if (currentQuizQuestionIndex >= quizQuestions.length) {

                showQuizResults();

                return;

            }

            const q = quizQuestions[currentQuizQuestionIndex];

            mainContent.innerHTML = `

                <div class="bg-white/80 backdrop-blur-sm p-8 rounded-2xl shadow-lg text-center">

                    <p class="text-gray-600 mb-2 font-semibold">Question ${currentQuizQuestionIndex + 1} of ${quizQuestions.length}</p>

                    <h2 class="text-2xl md:text-3xl font-bold mb-8 text-gray-800">${q.question}</h2>

                    <div class="grid grid-cols-1 md:grid-cols-2 gap-4">

                        ${q.options.map(option => `

                            <div class="quiz-option-container flex items-center bg-white rounded-lg shadow-md hover:shadow-lg transition-all duration-200 cursor-pointer" data-answer="${option.romaji}">

                                <span class="flex-grow p-4 text-lg text-pink-500 font-semibold text-center">${option.romaji}</span>

                                <button class="play-audio-btn p-4 text-gray-500 hover:text-pink-500 flex-shrink-0" data-text="${option.japanese}">${speakerIconSVG}</button>

                            </div>

                        `).join('')}

                    </div>

                </div>

            `;


            mainContent.querySelectorAll('.quiz-option-container').forEach(container => {

                container.addEventListener('click', handleQuizAnswer);

            });

        }


        function handleQuizAnswer(e) {

            const container = e.target.closest('.quiz-option-container');

            if (!container) return;


            if (e.target.closest('.play-audio-btn')) {

                playAudio(e.target.closest('.play-audio-btn').dataset.text, e);

                return;

            }


            mainContent.querySelectorAll('.quiz-option-container').forEach(c => c.style.pointerEvents = 'none');


            const selectedAnswer = container.dataset.answer;

            const correctAnswer = quizQuestions[currentQuizQuestionIndex].answer;

            

            if (selectedAnswer === correctAnswer) {

                quizScore++;

                container.classList.add('bg-green-400', 'text-white');

            } else {

                container.classList.add('bg-red-400', 'text-white');

            }


            setTimeout(() => {

                currentQuizQuestionIndex++;

                renderQuizQuestion();

            }, 1200);

        }

        

        function showQuizResults() {

            const scorePercentage = Math.round((quizScore / quizQuestions.length) * 100);

            document.getElementById('modal-score').textContent = `${scorePercentage}%`;

            

            let feedback = "Good effort! Keep practicing.";

            if (scorePercentage > 90) feedback = "Excellent! You're a natural!";

            else if (scorePercentage > 70) feedback = "Great job! You're getting the hang of it.";

            document.getElementById('modal-feedback').textContent = feedback;


            showQuizModal();

        }


        function showQuizModal() {

            quizModal.classList.remove('hidden');

            setTimeout(() => {

                quizModal.classList.remove('opacity-0');

                modalContent.classList.remove('scale-95', 'opacity-0');

                modalContent.classList.add('scale-100');

            }, 10);

        }


        function hideQuizModal() {

            modalContent.classList.remove('scale-100');

            modalContent.classList.add('scale-95', 'opacity-0');

            quizModal.classList.add('opacity-0');

            setTimeout(() => {

                quizModal.classList.add('hidden');

            }, 300);

        }


        // --- APP INITIALIZATION ---

        document.addEventListener('DOMContentLoaded', () => {

            // Initial load of voices

            loadVoices();

            // The 'voiceschanged' event is fired when the list of voices has been loaded and is ready.

            if (speechSynthesis.onvoiceschanged !== undefined) {

                speechSynthesis.onvoiceschanged = loadVoices;

            }

            showLearnView();

        });
